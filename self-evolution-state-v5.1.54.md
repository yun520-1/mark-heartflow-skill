```javascript
class HardProblemAnalyzer {
  analyzeProblemType(question) {
    const easyProblems = [
      'discrimination', 'integration', 'reporting',
      'attention', 'behavior control', 'wakefulness'
    ];
    
    const hardProblemMarkers = [
      'what it is like', 'subjective experience',
      'phenomenal feel', 'qualia', 'conscious experience'
    ];
    
    if (hardProblemMarkers.some(m => question.includes(m))) {
      return { type: 'hard', confidence: 0.95 };
    }
    if (easyProblems.some(p => question.includes(p))) {
      return { type: 'easy', confidence: 0.95 };
    }
    return { type: 'mixed', confidence: 0.7 };
  }
}
```
