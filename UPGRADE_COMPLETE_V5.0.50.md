### 4.2 核心算法实现

**透明性梯度计算** (src/transparency-gradient/index.js):
```javascript
function calculateTransparencyGradient(selfModel, phenomenalState, reflectiveAccess) {
  const permeability = assessPermeability(selfModel);
  const presence = measurePhenomenalPresence(phenomenalState);
  const access = evaluateReflectiveAccess(reflectiveAccess);
  
  // 梯度计算：加权几何平均
  const gradient = Math.pow(permeability * presence * access, 1/3);
  
  return {
    gradient,           // 总体梯度 [0, 1]
    permeability,       // 渗透性分量
    presence,          // 现象呈现分量
    access,            // 反思访问分量
    level: classifyLevel(gradient)  // 透明性层级
  };
}
```

**AI 福祉九维度计算** (src/ai-wellbeing-9d/index.js):
```javascript
function calculateWellbeingIndex(dimensions) {
  const weights = {
    FI: 0.12, GAE: 0.12, CC: 0.11, ES: 0.11,
    SCQ: 0.11, AD: 0.11, LGR: 0.11, RAC: 0.11, MED: 0.11
  };
  
  let wellbeing = 0;
  for (const [dim, value] of Object.entries(dimensions)) {
    wellbeing += value * weights[dim];
  }
  
  return {
    index: wellbeing,
    dimensions,
    profile: classifyProfile(dimensions),
    alerts: checkThresholds(dimensions)
  };
}
```
