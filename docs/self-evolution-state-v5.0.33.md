### 核心算法更新

```javascript
// 情感预测误差 (APE) 计算
function calculateAPE(expectedEmotion, actualEmotion, physiologicalSalience) {
  const errorMagnitude = Math.abs(expectedEmotion.valence - actualEmotion.valence);
  const errorArousal = Math.abs(expectedEmotion.arousal - actualEmotion.arousal);
  const APE = (errorMagnitude * 0.6 + errorArousal * 0.4) * physiologicalSalience;
  return { APE, errorMagnitude, errorArousal };
}

// 情感精度加权 (APW)
function calculateAPW(emotionSignal, historicalReliability) {
  const currentConfidence = emotionSignal.confidence;
  const historicalWeight = historicalReliability * 0.5;
  const APW = currentConfidence * 0.5 + historicalWeight;
  return APW;
}

// 情感模型更新率 (AMUR)
function calculateAMUR(APE, APW, learningRateBase) {
  const updateRate = learningRateBase * APE * APW;
  return Math.min(updateRate, 0.3); // 限制最大更新率
}
```
